def ajouter()
  @vBox1.pack_start(Gtk::Alignment.new(0, 0, 1, 1), :expand => false)
  @vBox1.add(@button)
  @vBox2.set_width_request($longListe)
  @vBox2.add(@titre)
  @vBox2.add(@vBoxBis0)
  @vBox2.add(@hBox2)
  add(@vBox1)
  pack_end(@vBox2, :expand => true, :fill => true, :padding => 0)
end

  def changerBox(maBox)
      @vBox2.each { |child|
         @vBox2.remove(child)
      }
      if(@i != 5)
        @vBox2.add(@titre)
      else
        @vBox2.add(@titre2)
      end
      @vBox2.add(maBox)
      if(@i != 5)
        @vBox2.add(@hBox2)
      else
        @i = 0
      end
      @vBox2.show_all
end
