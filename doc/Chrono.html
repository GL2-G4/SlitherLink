    class Chrono

    Presentation¶ ↑

    Chrono is a class that define a timer's behaviors

def getSec()
    hour = self.getTime.hour
    min = self.getTime.min
    sec = self.getTime.sec
    return hour*60*60 + min*60 + sec

def getTime
    if @timeStart != nil
        return @totalDuration + (Time.now() - @timeStart)
    else
        return @totalDuration
    end
end

def stop
    if @timeStart != nil
        @totalDuration += Time.now() - @timeStart
        @timeStart = nil
    end
    self
end

def to_s
    min = self.getTime.min.to_s
    sec = self.getTime.sec.to_s
    return "%02d:%02d" % [min, sec]
end
