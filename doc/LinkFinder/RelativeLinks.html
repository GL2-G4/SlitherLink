def to_s # Fonctionnelle
  i, sentence = 0, "#{@fileName} : "

  if self.hasLinks?
    @links.each { |repertory,relativeLink|
      sentence += "#{repertory} => #{relativeLink}"

      if i+1 < @links.length
        sentence += "|"
      end
      i += 1
    }
  else
    sentence += "ne possède aucun lien relatif connu"
  end
  return sentence
end
